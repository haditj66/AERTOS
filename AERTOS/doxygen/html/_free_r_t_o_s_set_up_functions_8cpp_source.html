18void prvInitialiseHeap(void)
19{
20 /* The Windows demo could create one large heap region, in which case it would
21 be appropriate to use heap_4. However, purely for demonstration purposes,
22 heap_5 is used instead, so start by defining some heap regions. No
23 initialisation is required when any other heap implementation is used. See
24 http://www.freertos.org/a00111.html for more information.
25
26 The xHeapRegions structure requires the regions to be defined in start address
27 order, so this just creates one big array, then populates the structure with
28 offsets into the array - with gaps in between and messy alignment just for test
29 purposes. */
30 static uint8_t ucHeap[configTOTAL_HEAP_SIZE];
31 volatile uint32_t ulAdditionalOffset = 19; /* Just to prevent 'condition is always true' warnings in configASSERT(). */
32 const HeapRegion_t xHeapRegions[] =
33 {
34 /* Start address with dummy offsets Size */
35 { ucHeap + 1, mainREGION_1_SIZE },
36 { ucHeap + 15 + mainREGION_1_SIZE, mainREGION_2_SIZE },
37 { ucHeap + 19 + mainREGION_1_SIZE + mainREGION_2_SIZE, mainREGION_3_SIZE },
38 { NULL, 0 }
39 };
40
41 /* Sanity check that the sizes and offsets defined actually fit into the
42 array. */
43 configASSERT((ulAdditionalOffset + mainREGION_1_SIZE + mainREGION_2_SIZE + mainREGION_3_SIZE) < configTOTAL_HEAP_SIZE);
44
45 /* Prevent compiler warnings when configASSERT() is not defined. */
46 (void)ulAdditionalOffset;
47
48 vPortDefineHeapRegions(xHeapRegions);
49}

102 void vApplicationMallocFailedHook(void)
103 {
104 /* vApplicationMallocFailedHook() will only be called if
105 configUSE_MALLOC_FAILED_HOOK is set to 1 in FreeRTOSConfig.h. It is a hook
106 function that will get called if a call to pvPortMalloc() fails.
107 pvPortMalloc() is called internally by the kernel whenever a task, queue,
108 timer or semaphore is created. It is also called by various parts of the
109 demo application. If heap_1.c, heap_2.c or heap_4.c is being used, then the
110 size of the heap available to pvPortMalloc() is defined by
111 configTOTAL_HEAP_SIZE in FreeRTOSConfig.h, and the xPortGetFreeHeapSize()
112 API function can be used to query the size of free heap space that remains
113 (although it does not provide information on how the remaining heap might be
114 fragmented). See http://www.freertos.org/a00111.html for more
115 information. */
116 vAssertCalled(__LINE__, __FILE__);
117 }
